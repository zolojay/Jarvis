SELECT
    LR.id                AS LRF_ID,
    LR.[number]          AS LRFNumber,
    LRS.id               AS LabRequestSampleID,
    S.id                 AS SampleID,
    S.[number]           AS SampleNumber,
    LRSTT.id            AS LabRequestSampleTestTypeID,
    TT.[id]             AS TestTypeID,
    TT.[test_type]      AS TestTypeName

FROM [dbo].[LabRequests] LR
-- Link to LabRequestSample
JOIN [dbo].[LabRequestSample] LRS
   ON LRS.lab_request_id = LR.id

-- Link to LabRequestSampleTestTypes
JOIN [dbo].[LabRequestSampleTestTypes] LRSTT
   ON LRSTT.lab_request_sample_id = LRS.id

-- Link to TestTypes (like bench, regen, etc.)
JOIN [dbo].[TestTypes] TT
   ON TT.id = LRSTT.test_type_id

-- Optionally link to the Samples table if you want sample info
JOIN [dbo].[Samples] S
   ON S.id = LRS.sample_id

-- Optional filter: only look at a specific LRF or all 
-- e.g. LRFNumber = 'LR-25-00005'
WHERE LR.[number] = 'LR-25-00005'
ORDER BY 
    LR.id,
    LRS.id,
    LRSTT.id;

SELECT
  LR.id                AS LRF_ID,
  LR.number            AS LRFNumber,
  LRS.id               AS LabRequestSampleID,
  S.id                 AS SampleID,
  S.number             AS SampleNumber,
  LRS_TT.id            AS LabRequestSampleTestTypeID,
  LRS_TT.test_type_id  AS TestTypeID,
  TT.test_type         AS TestTypeName,

  -- Potentially ReactorLoads or ReactorTests if they're linked by test_type_id
  RL.id                AS ReactorLoadID,
  RL.test_type_id      AS RL_TestTypeID, 
  RT.id                AS ReactorTestID,
  RT.test_condition_id,
  RTC.description,
  TCT.test_condition_type

FROM LabRequests LR
JOIN LabRequestSample LRS
   ON LRS.lab_request_id = LR.id
JOIN Samples S
   ON LRS.sample_id = S.id
JOIN LabRequestSampleTestTypes LRS_TT
   ON LRS_TT.lab_request_sample_id = LRS.id
JOIN TestTypes TT
   ON TT.id = LRS_TT.test_type_id

LEFT JOIN ReactorLoads RL
   ON RL.lab_request_id = LR.id
   AND RL.test_type_id = LRS_TT.test_type_id
LEFT JOIN ReactorTests RT
   ON RT.load_id = RL.id
LEFT JOIN ReactorTestConditions RTC
   ON RTC.id = RT.test_condition_id
LEFT JOIN TestConditionTypes TCT
   ON TCT.id = RTC.test_type_id

WHERE LR.id = 334  -- Just to filter to LRF 334 (LR-25-00005)
ORDER BY LR.id, LRS_TT.id;
